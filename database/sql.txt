CREATE TABLE `final`.`puestoelectivo` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(50) NULL DEFAULT NULL , 
`descripción` VARCHAR(50) NULL DEFAULT NULL , `estado` BOOLEAN NULL DEFAULT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

CREATE TABLE `final`.`partidos` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(50) NULL DEFAULT NULL ,
 `descripción` VARCHAR(50) NULL DEFAULT NULL ,
 `logo` VARCHAR(255) NULL DEFAULT NULL , `estado` BOOLEAN NULL DEFAULT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

 CREATE TABLE `final`.`elecciones` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(50) NULL DEFAULT NULL , 
 `fecha` DATE NULL DEFAULT NULL , `estado` BOOLEAN NULL DEFAULT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

 CREATE TABLE `final`.`ciudadanos` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `cedula` VARCHAR(100) NULL DEFAULT ,
  `nombre` VARCHAR(50) NULL DEFAULT NULL , `apellido` VARCHAR(50) NULL DEFAULT NULL ,
  `email` VARCHAR(50) NULL DEFAULT NULL , `estado` BOOLEAN NULL DEFAULT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

  CREATE TABLE `final`.`candidatos` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(50) NULL DEFAULT NULL ,
   `apellido` VARCHAR(50) NULL DEFAULT NULL , `partido` VARCHAR(50) NULL DEFAULT NULL , `puesto` VARCHAR(60) NULL DEFAULT NULL , 
   `foto` VARCHAR(255) NULL DEFAULT NULL , `estado` BOOLEAN NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

    CREATE TABLE `final`.`user` ( `id` INT(11) NOT NULL AUTO_INCREMENT , `userName` VARCHAR(150) NULL DEFAULT NULL , 
 `password` VARCHAR(100) NULL DEFAULT NULL , `firstName` VARCHAR(150) NULL DEFAULT NULL , 
 `lastName` VARCHAR(160) NULL DEFAULT NULL , `email` VARCHAR(255) NULL DEFAULT NULL , `photo` VARCHAR(255) NULL DEFAULT NULL , 
 PRIMARY KEY (`id`)) ENGINE = InnoDB;
